Å løkke gjennom arrays

En array har en lengde som er lik antall verdier lagret i arrayen:
   arrayen ['a', 'b', 'c', 'd'] har lengde 4.
   arrayen [123456] har lengde 1.
Dette kan vi bruke til å skrive ut alle verdiene i en løkke:
I løkken
   for (var i = plassen første element; i <= plassen til siste element; i++){
     // kode
   }
Vil plassen til første element være 0 og plassen til siste element være lengde-1. Løkken blir dermed
   for (var i = 0; i < arrayen.length; i++){
     // kode
   }

Vi kan for eksempel bruke dette til å skrive ut alle elementene i en array (ikke så spennende):
   var tallArray = [1, 3, 5, 8, 10, 2, 44, 0];
   for(var i = 0; i < tallArray.length; i++){
     console.log(tallArray[i]);
   }

Eller vi kan bruke det til å finne det største elementet i en array (litt mer spennende):
   var storst = tallArray[0];
   for(var i = 0; i < tallArray.length; i++){
     if (tallArray[i] > storst){
       storst = tallArray[i];
     }
   }
   console.log(storst, tallArray.indexOf(storst));

Koden for å finne største verdi i en array kan også utvides til en funksjon som tar inn arrayen og finner det største tallet:
   function finnStorst(tall){
     var storst = tall[0];
     for(var i = 0; i < tall.length; i++){
       if (tall[i] > storst){
         storst = tall[i];
       }
     }
     console.log(storst, tall.indexOf(storst));
   }
   var tallArray = [1, 2, 3, 7, 10, 2];
   finnStorst(tallArray) // evt finnStorst([1, 2, 3, 7, 10, 2])

Oppgaver
